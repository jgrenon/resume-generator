        var Q = require('q');

        function Renderer() {

        }

        Renderer.prototype.layout = function(layout) {
            this.layout = layout;
            return this;
        };

        Renderer.prototype.render = function(model) {
            this.model = model;
            return this;
        };

        Renderer.prototype.to = function(output) {
            var defer = Q.defer();
            this.output = output;

            Q.nextTick(function(){
                defer.resolve({
                    success:true
                });
            });

            return defer.promise;
        };

        /**
         * Factory method
         *
         * @returns {Renderer}
         */
        Renderer.load = function(){
            return new Renderer();
        };

        return Renderer;

    })();
